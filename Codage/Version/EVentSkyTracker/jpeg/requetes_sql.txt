# Afficher les candidates en fonction de l'id des evements, et de l'eventId des candidates
select c.*
from evenement e
inner join candidate c
on c.eventId = e.id
where e.id = c.eventId


# Afficher les evenements en fonction de l'id des evements, et de l'eventId des candidates
select e.*
from evenement e
inner join candidate c
on c.eventId = e.id
where e.id = c.eventId
